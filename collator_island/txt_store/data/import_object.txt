def _import_object(module_name: str, attr_name: Optional[str]):
    if not module_name:
        raise ValueError("module_name is required for custom dataset or collate function.")
    mod = import_module(module_name)
    if attr_name:
        if not hasattr(mod, attr_name):
            raise ImportError(f"Module '{module_name}' has no attribute '{attr_name}'")
        return getattr(mod, attr_name)
    return mod
