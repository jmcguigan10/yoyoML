
def plot_loss_curve(
    train_losses: list[float],
    val_losses: list[float],
    out_dir: Path,
    *,
    filename: str = "loss_curve.png",
    title: str = "Loss per epoch",
) -> None:
    import matplotlib.pyplot as plt

    epochs = list(range(1, max(len(train_losses), len(val_losses)) + 1))
    plt.figure()
    if train_losses:
        plt.plot(epochs[: len(train_losses)], train_losses, label="train")
    if val_losses:
        plt.plot(epochs[: len(val_losses)], val_losses, label="val")
    plt.xlabel("epoch")
    plt.ylabel("loss")
    plt.title(title)
    plt.legend()
    plt.tight_layout()
    plt.savefig(out_dir / filename, dpi=160)
    plt.close()
